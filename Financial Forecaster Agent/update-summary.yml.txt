name: Update Executive Summary

on:
  schedule:
    - cron: "0 */6 * * *"  # Runs every 6 hours
  push:
    branches:
      - main

jobs:
  update-summary:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout Repository
        uses: actions/checkout@v3

      - name: Install Dependencies
        run: pip install pandas requests transformers

      - name: Run Script to Generate Executive Summary
        run: python app.py

      - name: Push to Hugging Face
        run: |
          git config --global user.name "GitHub Actions"
          git config --global user.email "actions@github.com"
          git add executive-summary.html
          git commit -m "Updated executive summary" || echo "No changes to commit"
          git push
